CREATE TABLE salgrade (
	GRADE INT NOT NULL,
	LOSAL DECIMAL(8,2) NOT NULL,
	HISAL DECIMAL(8,2) NOT NULL,
	CONSTRAINT PRIMARY KEY (GRADE)
);

CREATE TABLE dept (
	DEPTNO INT NOT NULL,
	DNAME VARCHAR(14) NOT NULL,
	LOC VARCHAR(13) NOT NULL,
	CONSTRAINT PRIMARY KEY (DEPTNO)
);

CREATE TABLE emp (
	EMPNO INT NOT NULL,
	MGR INT,
	DEPTNO INT NOT NULL,
	ENAME VARCHAR(10) NOT NULL,
	JOB VARCHAR(9),
	HIREDATE DATE,
	SAL DECIMAL(8,2),
	COMM DECIMAL(8,2),
	CONSTRAINT PRIMARY KEY (EMPNO),
	CONSTRAINT emp_emp_fk FOREIGN KEY (MGR)
	REFERENCES emp (EMPNO),
	CONSTRAINT emp_dept_fk FOREIGN KEY (DEPTNO)
	REFERENCES dept (DEPTNO)
);
