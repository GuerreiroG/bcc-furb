LEITURAS:	
	MOV IN0,A; //0000000P
	MOV A,B; //MOVER A PARA B
	MOV IN1,A; //0000000C
	AND B,A; // AND ENTRE B E A
	MOV A,B;
	MOV IN2,A; //0000000M
	NOT A,A; //NEGA MARCHA
	MOV A,C;
	MOV IN3,A; //0000000E
	OR C,A; // OR ENTRE C E A
	AND B,A; //FAZ AND ENTRE B E A, COLOCANDO RESPOSTA EM A
	MOV A,OUT3;
JMP LEITURAS;
